import tkinter as tk
from tkinter import messagebox
import random

# Function to handle the game logic
def play_game(user_choice):
    choices = ["Rock", "Paper", "Scissors"]
    computer_choice = random.choice(choices)
    
    # Display the computer's choice
    computer_label.config(text=f"Computer chose: {computer_choice}")

    # Determine the outcome
    if user_choice == computer_choice:
        result = "It's a tie!"
    elif (user_choice == "Rock" and computer_choice == "Scissors") or \
         (user_choice == "Scissors" and computer_choice == "Paper") or \
         (user_choice == "Paper" and computer_choice == "Rock"):
        result = "You win!"
        update_score(1)
    else:
        result = "You lose!"
        update_score(-1)
    
    # Show the result in a popup
    messagebox.showinfo("Result", result)

def update_score(change):
    global score
    score += change
    score_label.config(text=f"Score: {score}")

# Create main window
root = tk.Tk()
root.title("Rock, Paper, Scissors Game")

# Set window size
root.geometry("400x400")
root.configure(bg="#f0f0f0")

# Create a frame for the game
frame = tk.Frame(root, bg="#ffffff", padx=20, pady=20)
frame.pack(pady=20)

# Initialize score
score = 0
score_label = tk.Label(frame, text=f"Score: {score}", font=("Helvetica", 16), bg="#ffffff")
score_label.pack(pady=10)

# Guidelines label
guidelines_label = tk.Label(frame, text="Choose Rock, Paper, or Scissors:", font=("Helvetica", 14), bg="#ffffff")
guidelines_label.pack(pady=10)

# Create and place buttons for Rock, Paper, and Scissors
rock_button = tk.Button(frame, text="Rock", width=10, command=lambda: play_game("Rock"), bg="#ff9999", font=("Helvetica", 12))
rock_button.pack(pady=5)

paper_button = tk.Button(frame, text="Paper", width=10, command=lambda: play_game("Paper"), bg="#99ff99", font=("Helvetica", 12))
paper_button.pack(pady=5)

scissors_button = tk.Button(frame, text="Scissors", width=10, command=lambda: play_game("Scissors"), bg="#9999ff", font=("Helvetica", 12))
scissors_button.pack(pady=5)

# Label to show the computer's choice
computer_label = tk.Label(frame, text="Computer's choice will appear here", font=("Helvetica", 12), bg="#ffffff")
computer_label.pack(pady=20)

# Run the Tkinter event loop
root.mainloop()
